### 1. Create New Branch

current state :: IDLE

choose number from 0 - 9

add new branch to tree

update branch index

reset node index

change mouse pointer to cross

change state to :: DRAWING


### 2. Add Node To Branch

current state :: DRAWING

get click position from single click

update node index

### 3. Finish Branch

current state :: DRAWING

double mouse click

change state :: IDLE

set mouse pointer to arrow

### 4. Delete Node

current state :: DRAWING

user press del/backspace

remove last node from branch

decrease node index

if 0 remove branch, change state to IDLE

### 5. Extend branch

current state :: DRAWING

draw current position from last node without adding points

### 6. Correct point

current state IDLE

check if close to node

if close enough change curser to circle

click down

get closest node index

change state to REPOSITION_NODE

click up

change state to IDLE

update node position

change mouse to arrow


### 7. Correct Branch

current state IDLE

check if close to branch line

if close enough change curser to hand

click down

get start position

change state to REPOSITION_BRANCH

move mouse

dx/dy = current position - start position

add dx/dy to each point to repaint

click up

add dx/dy to each point to repaint

update state to IDLE


### 8. Choose Branch

check if close to branch line

if close enough change curser to hand

double click

get current branch index

add highligh color maybe change alpha value & thickness

change state to EDIT_BRANCH


### 9. Delete Branch

current state EDIT_BRANCH

user press delete/backspace

remove branch

user clicks outside branch

remove highlight
change state to IDLE


#### EVENT - STATE TABLE

EVENT_KEY_DIGIT
	STATE_IDLE (obj.createBranch)
		->STATE_DRAWING

EVENT_KEY_DEL
	STATE_DRAWING (obj.deleteNode)
		->STATE_DRAWING
		->STATE_IDLE

	STATE_SELECTED_BRANCH (obj.deleteBranch)
		->STATE_IDLE

EVENT_CLICK_DOWN
	STATE_DRAWING (obj.extendBranch)
		->STATE_DRAWING

	STATE_OVER_BRANCH (obj.pickUpBranch)
		->STATE_REPOSITION_BRANCH

	STATE_OVER_NODE (obj.pickUpNode)
		->STATE_REPOSITION_NODE

EVENT_CLICK_UP
	STATE_REPOSITION_BRANCH (obj.putDownBranch)
		->STATE_IDLE

	STATE_REPOSITION_NODE (obj.putDownNode)
		->STATE_IDLE

EVENT_CLICK_DOUBLE
	STATE_DRAWING (obj.completeBranch)
		->STATE_IDLE

	STATE_OVER_BRANCH (obj.selectBranch)
		->STATE_SELECTED_BRANCH

EVENT_OVER_BRANCH
	STATE_IDLE
		->STATE_OVER_BRANCH

EVENT_OVER_NODE
	STATE_IDLE
		->STATE_OVER_NODE

EVENT_MOUSE_MOVE
	STATE_DRAWING (obj.stretchBranch)
		->STATE_DRAWING

	STATE_REPOSITION_BRANCH (obj.moveBranch)
		->STATE_REPOSITION_BRANCH

	STATE_REPOSITION_NODE (obj.moveNode)
		->STATE_REPOSITION_NODE

	STATE_OVER_BRANCH
		->STATE_IDLE

	STATE_OVER_NODE
		->STATE_IDLE
